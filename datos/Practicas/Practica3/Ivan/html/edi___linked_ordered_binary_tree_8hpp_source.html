256  virtual void removeItem()
257  {
258  assert(existsItem());
259  assert(checkInvariants() && ++_invariantCount);
260  edi::LinkedBinaryTreeNode<T> * replaceNode=0;
261  edi::LinkedBinaryTreeNode<T> * savePrev=0;
262  edi::LinkedBinaryTreeNode<T> * saveCurr=0;
263  bool replaceNodeFound=false;
264  if (not _curr->right()) //Node to remove has not rightSubTree
265  {
266  replaceNode=_curr->left();
267  replaceNodeFound=true;
268  }
269  else if (not _curr->left()) //Node to remove has not leftSubTree
270  {

272  replaceNodeFound=true;
273  }
274 
275  if (replaceNodeFound)
276  {
277  if (_prev==0)
278  setRoot(replaceNode);
279  else if (_prev->left()==_curr) //Node to remove is leftSubTree
280  _prev->setLeft(replaceNode);
281  else //Node to remove is rigthSubTree
282  _prev->setRight(replaceNode);
283  delete _curr; //free the node to remove
284  _curr=replaceNode; //_curr is the repalceNode
285  }
286  else //Node to remove has two subtrees.
287  {
288  savePrev=_prev;
289  saveCurr=_curr;
290  findInorderPrecessor(); //seek the highest of the leftSubTree
291  saveCurr->setItem(_curr->item());
